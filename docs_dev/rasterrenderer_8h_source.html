55 inline double BiCubicKernel(double x)
56 {
57  if ( x > 2.0 )
58  return 0.0;
59 
60  double a, b, c, d;
61  double xm1 = x - 1.0;
62  double xp1 = x + 1.0;
63  double xp2 = x + 2.0;
64 
65  a = ( xp2 <= 0.0 ) ? 0.0 : xp2 * xp2 * xp2;
66  b = ( xp1 <= 0.0 ) ? 0.0 : xp1 * xp1 * xp1;
67  c = ( x <= 0.0 ) ? 0.0 : x * x * x;
68  d = ( xm1 <= 0.0 ) ? 0.0 : xm1 * xm1 * xm1;
69 
70  return ( 0.16666666666666666667 * ( a - ( 4.0 * b ) + ( 6.0 * c ) - ( 4.0 * d ) ) );
71 };
